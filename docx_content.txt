## 🔒 3. Authentication & Bitwise RBAC Core## Custom Multi-Token Identifier Generation Algorithmic Flow
The Login ID is automatically assembled deterministically via memory string configurations to prevent database collisions:
$$\text{ID} = \text{Initial}_2 + \text{Char}_{10} + \text{Year}_4 + \text{Serial}_4$$ 
* $\text{Initial}_2$: Two-letter uppercase corporate identifier string (e.g., OI for Overlap Initial).
* $\text{Char}_{10}$: First 5 characters of the first name + first 5 characters of the last name (padded with X if shorter).
* $\text{Year}_4$: Calendar year of hire processing (e.g., 2026).
* $\text{Serial}_4$: Sequential sequence counter tracking users registered within that single year, zero-padded to 4 places.

import { FastifyInstance } from 'fastify';import { Buffer } from 'node:buffer';import { webcrypto } from 'node:crypto';import { ROLES } from '../../packages/shared-types/permissions';
export async function authRoutes(server: FastifyInstance) {
  server.post('/signup', async (request, reply) => {
    const { firstName, lastName, yearJoined, serialNum, email, password } = request.body as any;
    // Deterministic ID Construction
    const prefix = "OI";
    const fPart = firstName.replaceAll(/\s/g, '').slice(0, 5).padEnd(5, 'X').toUpperCase();
    const lPart = lastName.replaceAll(/\s/g, '').slice(0, 5).padEnd(5, 'X').toUpperCase();
    const serialStr = String(serialNum).padStart(4, '0');
    const generatedEmployeeCode = `${prefix}${fPart}${lPart}${yearJoined}${serialStr}`;
    // Cryptographic Password Hashing via Native WebCrypto PBKDF2
    const salt = webcrypto.getRandomValues(new Uint8Array(16));
    const baseKey = await webcrypto.subtle.importKey(
      "raw", new TextEncoder().encode(password), "PBKDF2", false, ["deriveBits", "deriveKey"]
    );
    const derivedKeyBuffer = await webcrypto.subtle.deriveBits(
      { name: "PBKDF2", salt, iterations: 100000, hash: "SHA-256" }, baseKey, 256
    );
    const passwordHash = Buffer.from(derivedKeyBuffer).toString('hex');
    const saltHex = Buffer.from(salt).toString('hex');
    // Default Role Allocation
    const roleMask = ROLES.EMPLOYEE;
    return reply.status(201).send({ employeeCode: generatedEmployeeCode, hash: passwordHash, salt: saltHex, roleMask });
  });
}

## apps/backend-api/routes/attendance.ts
import { FastifyInstance } from 'fastify';import { pool } from '../../packages/database/pool';
interface AttendanceTick {
  employeeId: string;
  type: 'IN' | 'OUT';
  timestamp: Date;
}
export async function attendanceRoutes(server: FastifyInstance) {
  // Pre-allocated High-Performance Memory-Lease Queue Array for Micro-Batching
  let memoryQueue: AttendanceTick[] = [];
  const BATCH_THRESHOLD = 50;
  const FLUSH_INTERVAL_MS = 100;
  async function flushBatch() {
    if (memoryQueue.length === 0) return;
    const processingBatch = [...memoryQueue];
    memoryQueue = [];
    const client = await pool.connect();
    try {
      await client.query('BEGIN');
      for (const tick of processingBatch) {
        const dateStr = tick.timestamp.toISOString().split('T')[0];
        if (tick.type === 'IN') {
          await client.query(`
            INSERT INTO attendance_ledger (employee_id, work_date, check_in)
            VALUES ($1, $2, $3)
            ON CONFLICT (employee_id, work_date) DO NOTHING`,
            [tick.employeeId, dateStr, tick.timestamp]
          );
          await client.query(
            `UPDATE employees SET current_status = 'PRESENT' WHERE id = $1`,
            [tick.employeeId]
          );
        } else {
          await client.query(`
            UPDATE attendance_ledger 
            SET check_out = $3,
                work_hours = EXTRACT(EPOCH FROM ($3 - check_in))/3600 - break_hours
            WHERE employee_id = $1 AND work_date = $2`,
            [tick.employeeId, dateStr, tick.timestamp]
          );
          await client.query(
            `UPDATE employees SET current_status = 'ABSENT' WHERE id = $1`,
            [tick.employeeId]
          );
        }
      }
      await client.query('COMMIT');
    } catch (err) {
      await client.query('ROLLBACK');
    } finally {
      client.release();
    }
  }
  // Interval-Based Fallback Flush
  setInterval(flushBatch, FLUSH_INTERVAL_MS);
  server.post('/tick', async (request, reply) => {
    const { employeeId, type } = request.body as { employeeId: string; type: 'IN' | 'OUT' };
    
    // In-memory queue storage bypasses heavy database row locks immediately
    memoryQueue.push({ employeeId, type, timestamp: new Date() });
    if (memoryQueue.length >= BATCH_THRESHOLD) {
      process.nextTick(flushBatch);
    }
    return reply.status(202).send({ status: 'Processing tick in memory buffer queue' });
  });
}

'use client';import React, { useState } from 'react';
interface DayCell {
  dateString: string;
  dayOfMonth: number;
  status: 'PRESENT' | 'ABSENT' | 'LEAVE' | 'REST_DAY';
}
export function InteractiveCalendar({ year, month, historicalLogs }: { year: number; month: number; historicalLogs: Record<string, string> }) {
  const [selectedDates, setSelectedDates] = useState<string[]>([]);
  
  // High-density pure JS computing of specific month matrices without moment/date-fns
  const totalDays = new Date(year, month + 1, 0).getDate();
  const firstDayIndex = new Date(year, month, 1).getDay();
  
  const calendarGrid: DayCell[] = [];
  
  // Pad preceding offsets
  for (let i = 0; i < firstDayIndex; i++) {
    calendarGrid.push({ dateString: '', dayOfMonth: 0, status: 'REST_DAY' });
  }
  // Inject structural records
  for (let day = 1; day <= totalDays; day++) {
    const dString = `${year}-${String(month + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
    const logStatus = (historicalLogs[dString] || 'ABSENT') as any;
    calendarGrid.push({ dateString: dString, dayOfMonth: day, status: logStatus });
  }
  const handleSelection = (dateStr: string) => {
    if (!dateStr) return;
    setSelectedDates(prev => 
      prev.includes(dateStr) ? prev.filter(d => d !== dateStr) : [...prev, dateStr]
    );
  };
  return (
    <div className="w-full max-w-4xl p-6 bg-slate-900 text-white rounded-xl shadow-2xl border border-slate-800">
      <div className="grid grid-cols-7 gap-2 mb-4 text-center font-bold text-slate-400">
        {['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'].map(d => <div key={d} className="p-2 text-xs uppercase tracking-wider">{d}</div>)}
      </div>
      <div className="grid grid-cols-7 gap-2">
        {calendarGrid.map((cell, idx) => {
          if (cell.dayOfMonth === 0) return <div key={`empty-${idx}`} className="bg-slate-950/40 rounded-lg p-4" />;
          
          const isSelected = selectedDates.includes(cell.dateString);
          const colorMap = {
            PRESENT: 'border-l-4 border-emerald-500 bg-emerald-950/20 text-emerald-200',
            ABSENT: 'border-l-4 border-amber-500 bg-amber-950/20 text-amber-200',
            LEAVE: 'border-l-4 border-sky-500 bg-sky-950/20 text-sky-200',
            REST_DAY: 'bg-slate-800 text-slate-400'
          };
          return (
            <button
              key={cell.dateString}
              onClick={() => handleSelection(cell.dateString)}
              className={`p-3 text-left h-20 rounded-lg transition-all border font-medium ${colorMap[cell.status]} ${
                isSelected ? 'ring-2 ring-indigo-500 scale-95 border-indigo-400' : 'border-slate-800 hover:border-slate-700'
              }`}
            >
              <span className="block text-sm font-bold text-white">{cell.dayOfMonth}</span>
              <span className="block text-[10px] mt-1 font-semibold tracking-wide uppercase opacity-80">{cell.status}</span>
            </button>
          );
        })}
      </div>
    </div>
  );
}

## 🧮 6. Automated Payroll Assembly Engine## packages/payroll-engine/calculator.ts
export interface SalaryComponents {
  basic: number;
  hra: number;
  standardAllowance: number;
  performanceBonus: number;
  leaveTravelAllowance: number;
  fixedAllowance: number;
  employeePF: number;
  employerPF: number;
  professionalTax: number;
  grossEarnings: number;
  totalDeductions: number;
  netTakeHome: number;
}
export function compileSalaryStructure(monthlyWage: number, totalPayableDays: number, totalWorkingDays: number): SalaryComponents {
  // Prorate salary structurally matching attendance records
  const performanceRatio = totalPayableDays / totalWorkingDays;
  const effectiveBaseWage = monthlyWage * performanceRatio;
  // Exact Business Logic Assertions
  const basic = Math.round(effectiveBaseWage * 0.50);
  const hra = Math.round(basic * 0.50);
  const standardAllowance = 4167 * performanceRatio;
  const performanceBonus = Math.round(basic * 0.0833);
  const leaveTravelAllowance = Math.round(basic * 0.08333);
  // Subtotal current allocations to evaluate the dynamic Fixed Allowance residue safely
  const aggregatedAllocated = basic + hra + standardAllowance + performanceBonus + leaveTravelAllowance;
  const fixedAllowance = Math.max(0, effectiveBaseWage - aggregatedAllocated);
  const grossEarnings = basic + hra + standardAllowance + performanceBonus + leaveTravelAllowance + fixedAllowance;
  // Deductions Layer
  const employeePF = Math.round(basic * 0.12);
  const employerPF = Math.round(basic * 0.12);
  const professionalTax = effectiveBaseWage > 15000 ? 200 : 0;
  const totalDeductions = employeePF + professionalTax;
  const netTakeHome = grossEarnings - totalDeductions;
  return {
    basic, hra, standardAllowance, performanceBonus, leaveTravelAllowance,
    fixedAllowance, employeePF, employerPF, professionalTax,
    grossEarnings, totalDeductions, netTakeHome
  };
}

Custom Multi-Token Identifier Generation Algorithmic Flow
The Login ID is automatically assembled deterministically via memory string configurations to prevent database collisions:
\(\text{ID}=\text{Initial}_{2}+\text{Char}_{10}+\text{Year}_{4}+\text{Serial}_{4}\)
\(\text{Initial}_{2}\): Two-letter uppercase corporate identifier string (e.g., OI for Overlap Initial).
\(\text{Char}_{10}\): First 5 characters of the first name + first 5 characters of the last name (padded with X if shorter).
\(\text{Year}_{4}\): Calendar year of hire processing (e.g., 2026).
\(\text{Serial}_{4}\): Sequential sequence counter tracking users registered within that single year, zero-padded to 4 places.

import { FastifyInstance } from 'fastify';
import { Buffer } from 'node:buffer';
import { webcrypto } from 'node:crypto';
import { ROLES } from '../../packages/shared-types/permissions';
export async function authRoutes(server: FastifyInstance) {
  server.post('/signup', async (request, reply) => {
    const { firstName, lastName, yearJoined, serialNum, email, password } = request.body as any;
    // Deterministic ID Construction
    const prefix = "OI";
    const fPart = firstName.replaceAll(/\s/g, '').slice(0, 5).padEnd(5, 'X').toUpperCase();
    const lPart = lastName.replaceAll(/\s/g, '').slice(0, 5).padEnd(5, 'X').toUpperCase();
    const serialStr = String(serialNum).padStart(4, '0');
    const generatedEmployeeCode = `${prefix}${fPart}${lPart}${yearJoined}${serialStr}`;
    // Cryptographic Password Hashing via Native WebCrypto PBKDF2
    const salt = webcrypto.getRandomValues(new Uint8Array(16));
    const baseKey = await webcrypto.subtle.importKey(
      "raw", new TextEncoder().encode(password), "PBKDF2", false, ["deriveBits", "deriveKey"]
    );
    const derivedKeyBuffer = await webcrypto.subtle.deriveBits(
      { name: "PBKDF2", salt, iterations: 100000, hash: "SHA-256" }, baseKey, 256
    );
    const passwordHash = Buffer.from(derivedKeyBuffer).toString('hex');
    const saltHex = Buffer.from(salt).toString('hex');
    // Default Role Allocation
    const roleMask = ROLES.EMPLOYEE;
    return reply.status(201).send({ employeeCode: generatedEmployeeCode, hash: passwordHash, salt: saltHex, roleMask });
  });
}

import { FastifyInstance } from 'fastify';
import { pool } from '../../packages/database/pool';
interface AttendanceTick {
  employeeId: string;
  type: 'IN' | 'OUT';
  timestamp: Date;
}
export async function attendanceRoutes(server: FastifyInstance) {
  // Pre-allocated High-Performance Memory-Lease Queue Array for Micro-Batching
  let memoryQueue: AttendanceTick[] = [];
  const BATCH_THRESHOLD = 50;
  const FLUSH_INTERVAL_MS = 100;
  async function flushBatch() {
    if (memoryQueue.length === 0) return;
    const processingBatch = [...memoryQueue];
    memoryQueue = [];
    const client = await pool.connect();
    try {
      await client.query('BEGIN');
      for (const tick of processingBatch) {
        const dateStr = tick.timestamp.toISOString().split('T')[0];
        if (tick.type === 'IN') {
          await client.query(`
            INSERT INTO attendance_ledger (employee_id, work_date, check_in)
            VALUES ($1, $2, $3)
            ON CONFLICT (employee_id, work_date) DO NOTHING`,
            [tick.employeeId, dateStr, tick.timestamp]
          );
          await client.query(
            `UPDATE employees SET current_status = 'PRESENT' WHERE id = $1`,
            [tick.employeeId]
          );
        } else {
          await client.query(`
            UPDATE attendance_ledger 
            SET check_out = $3,
                work_hours = EXTRACT(EPOCH FROM ($3 - check_in))/3600 - break_hours
            WHERE employee_id = $1 AND work_date = $2`,
            [tick.employeeId, dateStr, tick.timestamp]
          );
          await client.query(
            `UPDATE employees SET current_status = 'ABSENT' WHERE id = $1`,
            [tick.employeeId]
          );
        }
      }
      await client.query('COMMIT');
    } catch (err) {
      await client.query('ROLLBACK');
    } finally {
      client.release();
    }
  }
  // Interval-Based Fallback Flush
  setInterval(flushBatch, FLUSH_INTERVAL_MS);
  server.post('/tick', async (request, reply) => {
    const { employeeId, type } = request.body as { employeeId: string; type: 'IN' | 'OUT' };
    
    // In-memory queue storage bypasses heavy database row locks immediately
    memoryQueue.push({ employeeId, type, timestamp: new Date() });
    if (memoryQueue.length >= BATCH_THRESHOLD) {
      process.nextTick(flushBatch);
    }
    return reply.status(202).send({ status: 'Processing tick in memory buffer queue' });
  });
}

import { FastifyInstance } from 'fastify';
import { Buffer } from 'node:buffer';
import { webcrypto } from 'node:crypto';
import { ROLES } from '../../packages/shared-types/permissions';
export async function authRoutes(server: FastifyInstance) {
  server.post('/signup', async (request, reply) => {
    const { firstName, lastName, yearJoined, serialNum, email, password } = request.body as any;
    // Deterministic ID Construction
    const prefix = "OI";
    const fPart = firstName.replaceAll(/\s/g, '').slice(0, 5).padEnd(5, 'X').toUpperCase();
    const lPart = lastName.replaceAll(/\s/g, '').slice(0, 5).padEnd(5, 'X').toUpperCase();
    const serialStr = String(serialNum).padStart(4, '0');
    const generatedEmployeeCode = `${prefix}${fPart}${lPart}${yearJoined}${serialStr}`;
    // Cryptographic Password Hashing via Native WebCrypto PBKDF2
    const salt = webcrypto.getRandomValues(new Uint8Array(16));
    const baseKey = await webcrypto.subtle.importKey(
      "raw", new TextEncoder().encode(password), "PBKDF2", false, ["deriveBits", "deriveKey"]
    );
    const derivedKeyBuffer = await webcrypto.subtle.deriveBits(
      { name: "PBKDF2", salt, iterations: 100000, hash: "SHA-256" }, baseKey, 256
    );
    const passwordHash = Buffer.from(derivedKeyBuffer).toString('hex');
    const saltHex = Buffer.from(salt).toString('hex');
    // Default Role Allocation
    const roleMask = ROLES.EMPLOYEE;
    return reply.status(201).send({ employeeCode: generatedEmployeeCode, hash: passwordHash, salt: saltHex, roleMask });
  });
}

import { FastifyInstance } from 'fastify';
import { pool } from '../../packages/database/pool';
interface AttendanceTick {
  employeeId: string;
  type: 'IN' | 'OUT';
  timestamp: Date;
}
export async function attendanceRoutes(server: FastifyInstance) {
  // Pre-allocated High-Performance Memory-Lease Queue Array for Micro-Batching
  let memoryQueue: AttendanceTick[] = [];
  const BATCH_THRESHOLD = 50;
  const FLUSH_INTERVAL_MS = 100;
  async function flushBatch() {
    if (memoryQueue.length === 0) return;
    const processingBatch = [...memoryQueue];
    memoryQueue = [];
    const client = await pool.connect();
    try {
      await client.query('BEGIN');
      for (const tick of processingBatch) {
        const dateStr = tick.timestamp.toISOString().split('T')[0];
        if (tick.type === 'IN') {
          await client.query(`
            INSERT INTO attendance_ledger (employee_id, work_date, check_in)
            VALUES ($1, $2, $3)
            ON CONFLICT (employee_id, work_date) DO NOTHING`,
            [tick.employeeId, dateStr, tick.timestamp]
          );
          await client.query(
            `UPDATE employees SET current_status = 'PRESENT' WHERE id = $1`,
            [tick.employeeId]
          );
        } else {
          await client.query(`
            UPDATE attendance_ledger 
            SET check_out = $3,
                work_hours = EXTRACT(EPOCH FROM ($3 - check_in))/3600 - break_hours
            WHERE employee_id = $1 AND work_date = $2`,
            [tick.employeeId, dateStr, tick.timestamp]
          );
          await client.query(
            `UPDATE employees SET current_status = 'ABSENT' WHERE id = $1`,
            [tick.employeeId]
          );
        }
      }
      await client.query('COMMIT');
    } catch (err) {
      await client.query('ROLLBACK');
    } finally {
      client.release();
    }
  }
  // Interval-Based Fallback Flush
  setInterval(flushBatch, FLUSH_INTERVAL_MS);
  server.post('/tick', async (request, reply) => {
    const { employeeId, type } = request.body as { employeeId: string; type: 'IN' | 'OUT' };
    
    // In-memory queue storage bypasses heavy database row locks immediately
    memoryQueue.push({ employeeId, type, timestamp: new Date() });
    if (memoryQueue.length >= BATCH_THRESHOLD) {
      process.nextTick(flushBatch);
    }
    return reply.status(202).send({ status: 'Processing tick in memory buffer queue' });
  });
}

'use client';
import React, { useState } from 'react';
interface DayCell {
  dateString: string;
  dayOfMonth: number;
  status: 'PRESENT' | 'ABSENT' | 'LEAVE' | 'REST_DAY';
}
export function InteractiveCalendar({ year, month, historicalLogs }: { year: number; month: number; historicalLogs: Record<string, string> }) {
  const [selectedDates, setSelectedDates] = useState<string[]>([]);
  
  // High-density pure JS computing of specific month matrices without moment/date-fns
  const totalDays = new Date(year, month + 1, 0).getDate();
  const firstDayIndex = new Date(year, month, 1).getDay();
  
  const calendarGrid: DayCell[] = [];
  
  // Pad preceding offsets
  for (let i = 0; i < firstDayIndex; i++) {
    calendarGrid.push({ dateString: '', dayOfMonth: 0, status: 'REST_DAY' });
  }
  // Inject structural records
  for (let day = 1; day <= totalDays; day++) {
    const dString = `${year}-${String(month + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
    const logStatus = (historicalLogs[dString] || 'ABSENT') as any;
    calendarGrid.push({ dateString: dString, dayOfMonth: day, status: logStatus });
  }
  const handleSelection = (dateStr: string) => {
    if (!dateStr) return;
    setSelectedDates(prev => 
      prev.includes(dateStr) ? prev.filter(d => d !== dateStr) : [...prev, dateStr]
    );
  };
  return (
    <div className="w-full max-w-4xl p-6 bg-slate-900 text-white rounded-xl shadow-2xl border border-slate-800">
      <div className="grid grid-cols-7 gap-2 mb-4 text-center font-bold text-slate-400">
        {['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'].map(d => <div key={d} className="p-2 text-xs uppercase tracking-wider">{d}</div>)}
      </div>
      <div className="grid grid-cols-7 gap-2">
        {calendarGrid.map((cell, idx) => {
          if (cell.dayOfMonth === 0) return <div key={`empty-${idx}`} className="bg-slate-950/40 rounded-lg p-4" />;
          
          const isSelected = selectedDates.includes(cell.dateString);
          const colorMap = {
            PRESENT: 'border-l-4 border-emerald-500 bg-emerald-950/20 text-emerald-200',
            ABSENT: 'border-l-4 border-amber-500 bg-amber-950/20 text-amber-200',
            LEAVE: 'border-l-4 border-sky-500 bg-sky-950/20 text-sky-200',
            REST_DAY: 'bg-slate-800 text-slate-400'
          };
          return (
            <button
              key={cell.dateString}
              onClick={() => handleSelection(cell.dateString)}
              className={`p-3 text-left h-20 rounded-lg transition-all border font-medium ${colorMap[cell.status]} ${
                isSelected ? 'ring-2 ring-indigo-500 scale-95 border-indigo-400' : 'border-slate-800 hover:border-slate-700'
              }`}
            >
              <span className="block text-sm font-bold text-white">{cell.dayOfMonth}</span>
              <span className="block text-[10px] mt-1 font-semibold tracking-wide uppercase opacity-80">{cell.status}</span>
            </button>
          );
        })}
      </div>
    </div>
  );
}

export interface SalaryComponents {
  basic: number;
  hra: number;
  standardAllowance: number;
  performanceBonus: number;
  leaveTravelAllowance: number;
  fixedAllowance: number;
  employeePF: number;
  employerPF: number;
  professionalTax: number;
  grossEarnings: number;
  totalDeductions: number;
  netTakeHome: number;
}
export function compileSalaryStructure(monthlyWage: number, totalPayableDays: number, totalWorkingDays: number): SalaryComponents {
  // Prorate salary structurally matching attendance records
  const performanceRatio = totalPayableDays / totalWorkingDays;
  const effectiveBaseWage = monthlyWage * performanceRatio;
  // Exact Business Logic Assertions
  const basic = Math.round(effectiveBaseWage * 0.50);
  const hra = Math.round(basic * 0.50);
  const standardAllowance = 4167 * performanceRatio;
  const performanceBonus = Math.round(basic * 0.0833);
  const leaveTravelAllowance = Math.round(basic * 0.08333);
  // Subtotal current allocations to evaluate the dynamic Fixed Allowance residue safely
  const aggregatedAllocated = basic + hra + standardAllowance + performanceBonus + leaveTravelAllowance;
  const fixedAllowance = Math.max(0, effectiveBaseWage - aggregatedAllocated);
  const grossEarnings = basic + hra + standardAllowance + performanceBonus + leaveTravelAllowance + fixedAllowance;
  // Deductions Layer
  const employeePF = Math.round(basic * 0.12);
  const employerPF = Math.round(basic * 0.12);
  const professionalTax = effectiveBaseWage > 15000 ? 200 : 0;
  const totalDeductions = employeePF + professionalTax;
  const netTakeHome = grossEarnings - totalDeductions;
  return {
    basic, hra, standardAllowance, performanceBonus, leaveTravelAllowance,
    fixedAllowance, employeePF, employerPF, professionalTax,
    grossEarnings, totalDeductions, netTakeHome
  };
}

async function verifyPass(input: string, salt: string, targetHash: string): Promise<boolean> {
  const baseKey = await webcrypto.subtle.importKey(
    "raw", new TextEncoder().encode(input), "PBKDF2", false, ["deriveBits"]
  );
  const derived = await webcrypto.subtle.deriveBits(
    { name: "PBKDF2", salt: Buffer.from(salt, 'hex'), iterations: 100000, hash: "SHA-256" }, baseKey, 256
  );
  return Buffer.from(derived).toString('hex') === targetHash;
}
